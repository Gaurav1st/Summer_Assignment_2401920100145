class Solution {
public:
    int removeDuplicates(vector<int>& nums) {
        
        int wap=1;
        int idx=1;
        while(idx<nums.size())
        {
            if(nums[idx]==nums[idx-1])
            {
                idx++;
            }
            else
            {
                nums[wap]=nums[idx];
                wap++;
                idx++;
            }
        }
        return wap;
    }
};
